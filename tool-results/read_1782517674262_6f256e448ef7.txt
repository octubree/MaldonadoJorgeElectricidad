   745→            document.getElementById('message-error').style.display = 'block';
   746→            isValid = false;

   752→            isValid = false;

   755→          if (contactWhatsapp.checked) {

   757→            const number = document.getElementById('whatsapp_number').value.trim();
   758→            const whatsappError = document.getElementById('whatsapp-error');

   760→            if (country === 'uy') {
   761→              phoneRegex = /^09[1-9]\d{6}$/;

   763→                whatsappError.textContent = 'El número de Uruguay debe tener el formato 09X XXX XXX (9 dígitos).';

   765→                isValid = false;
   766→              }
